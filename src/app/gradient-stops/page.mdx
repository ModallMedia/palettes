```php
// PHP (using cURL)
function requestColorSchemes() {
    $color = "#ff0000";  // Replace with your color value
    $count = 5;  // Replace with your count value
    $url = "https://palettespro.com/api/v1/gradient-stops?color=" . urlencode($color) . "&count=" . urlencode($count);

    $curl = curl_init($url);
    curl_setopt($curl, CURLOPT_RETURNTRANSFER, true);

    $response = curl_exec($curl);
    if ($response === false) {
        curl_close($curl);
        return null;
    }

    curl_close($curl);
    return json_decode($response);
}

// Call the function
$colorSchemes = requestColorSchemes();
print_r($colorSchemes);
```
